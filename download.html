Get System Rules

The System Rules library is available from the Maven repository.

Download

  • system-rules-1.3.1.jar
  • system-rules-1.3.1-javadoc.jar
  • system-rules-1.3.1-sources.jar

Use in Maven

<dependency>
    <groupId>com.github.stefanbirkner</groupId>
    <artifactId>system-rules</artifactId>
    <version>1.3.1</version>
</dependency>

Use in Buildr

'com.github.stefanbirkner:system-rules:jar:1.3.1'

Use in Ivy

<dependency org="com.github.stefanbirkner" name="system-rules" rev="1.3.1" >
    <artifact name="system-rules" type="jar"/>
</dependency>

Use in Groovy Grape

@Grapes(
	@Grab(group='com.github.stefanbirkner', module='system-rules', version='1.3.1')
)

Use in Grails

compile 'com.github.stefanbirkner:system-rules:1.3.1'
